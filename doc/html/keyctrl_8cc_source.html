00052 bool Keyboard_Controller::key_decoded () {
00053   bool done = false;
00054   
00055   // keys of MF II keyboards start sending one of two prefix bytes
00056   if (code == prefix1 || code == prefix2) { 
00057     prefix = code;
00058     return false;
00059   }
00060   
00061   // releasing a key is important for the modifier keys (SHIFT, CTRL, ALT, ...)
00062   // break codes from other keys are ignored
00063   if (code & break_bit) {
00064     // break code of a key is equal to the make code but additionaly the 
00065     // break_bit is set
00066     code &= ~break_bit;
00067     switch (code) {
00068       case 42:
00069       case 54:
00070         gather.shift (false);
00071         break;
00072       case 56:
00073         if (prefix == prefix1)
00074           gather.alt_right (false);
00075         else
00076           gather.alt_left (false);
00077         break;
00078       case 29:
00079         if (prefix == prefix1)
00080           gather.ctrl_right (false);
00081         else
00082           gather.ctrl_left (false);
00083         break;
00084     }
00085     
00086     // a prefix is only valid for the following code
00087     prefix = 0;
00088     
00089     // returns false since the break codes gives no additional information
00090     return false;
00091   }
00092   
00093   // a key has been pressed, if it was a modifier key like SHIFT, ALT, NUM_LOCK
00094   // eg. only the internal state is changed. Return value 'false' shows that 
00095   // the input is not finished yet. If other keys have been pressed the ASCII 
00096   // and Scancode is read and 'ture' is returned.
00097   switch (code) {
00098     case 42:
00099     case 54:
00100       gather.shift (true);
00101       break;
00102     case 56:
00103       if (prefix == prefix1)
00104         gather.alt_right (true);
00105       else
00106         gather.alt_left (true);
00107         break;
00108     case 29:
00109       if (prefix == prefix1)
00110         gather.ctrl_right (true);
00111       else
00112         gather.ctrl_left (true);
00113       break;
00114     case 58:
00115       gather.caps_lock (!gather.caps_lock ());
00116       set_led (caps_lock, gather.caps_lock());
00117       break;
00118     case 70:
00119       gather.scroll_lock (!gather.scroll_lock ());
00120       set_led (scroll_lock, gather.scroll_lock());
00121       break;
00122     case 69: // NUM_LOCK or BREAK ?
00123       if (gather.ctrl_left ()) {        // BREAK
00124         get_ascii_code ();
00125         done = true;
00126       } else {                          // NUM_LOCK
00127         gather.num_lock (!gather.num_lock());
00128         set_led (num_lock, gather.num_lock ());
00129       }
00130       break;
00131     default:                            // other keys
00132        // get ASCII codes from table, DONE
00133       get_ascii_code ();
00134       done = true;
00135   }
00136   
00137     // a prefix is only valid for the following code
00138   prefix = 0;
00139   
00140   if (done)
00141     return true;
00142   else
00143     return false;
00144 }

00149 void Keyboard_Controller::get_ascii_code () {
00150   if (code == 53 && prefix == prefix1) { // special case scancode 53
00151     gather.ascii ('/');
00152     gather.scancode (Key::scan::div);
00153   } else if (gather.num_lock () && !prefix && code>=71 && code<=83) {
00154     gather.ascii (asc_num_tab[code-71]);
00155     gather.scancode (scan_num_tab[code-71]);
00156   } else if (gather.alt_right ()) {
00157     gather.ascii (alt_tab[code]);
00158     gather.scancode (code);
00159   } else if (gather.shift ()) {
00160     gather.ascii (shift_tab[code]);
00161     gather.scancode (code);
00162   } else if (gather.caps_lock ()) {
00163     //Linux CapsLock, nur Buchstaben gross
00164     if ((code>=16 && code<=26) || (code>=30 && code<=40) || (code>=44 && code<=50)) {
00165       gather.ascii (shift_tab[code]);
00166       gather.scancode (code);
00167     } else {
00168       gather.ascii (normal_tab[code]);
00169       gather.scancode (code);
00170     }
00171   } else {
00172     gather.ascii (normal_tab[code]);
00173     gather.scancode (code);
00174   }
00175 }
